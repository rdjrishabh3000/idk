guessing_game.py

import random
import os

HIGHSCORE_FILE = "highscore.txt"

def load_high_score():
    if os.path.exists(HIGHSCORE_FILE):
        try:
            with open(HIGHSCORE_FILE, "r") as f:
                return int(f.read().strip())
        except ValueError:
            return None
    return None

def save_high_score(score):
    with open(HIGHSCORE_FILE, "w") as f:
        f.write(str(score))

def get_difficulty():
    print("\nSelect Difficulty Level:")
    print("1. Easy   (1-50,  10 attempts)")
    print("2. Medium (1-100, 7 attempts)")
    print("3. Hard   (1-200, 5 attempts)")
    
    while True:
        choice = input("Enter choice (1, 2, or 3): ").strip()
        if choice == "1": return 50, 10
        elif choice == "2": return 100, 7
        elif choice == "3": return 200, 5
        print("❌ Invalid choice.")

def play_game():
    print("=========================================")
    print("🎯 WELCOME TO THE NUMBER GUESSING GAME! 🎯")
    print("=========================================")
    
    high_score = load_high_score()
    if high_score is not None:
        print(f"🏆 Current High Score Record: {high_score} attempts")
    else:
        print("🏆 No high score set yet.")
        
    max_num, max_attempts = get_difficulty()
    secret_number = random.randint(1, max_num)
    attempts = 0

    while attempts < max_attempts:
        try:
            guess = int(input(f"\n[Attempt {attempts + 1}/{max_attempts}] Enter guess: "))
            attempts += 1
            if guess < secret_number: print("📉 Too low!")
            elif guess > secret_number: print("📈 Too high!")
            else:
                print(f"\n🎉 Correct in {attempts} attempts!")
                if high_score is None or attempts < high_score:
                    save_high_score(attempts)
                break
        except ValueError:
            print("❌ Invalid input.")
    else:
        print(f"\n💥 Game Over! Number was {secret_number}.")

    input("\nPress Enter to exit...")

if __name__ == '__main__':
    play_game()
